еҹәжң¬дҪҝз”Ё вҖӢ

unplugin-auto-import вҖӢ

йЎ№зӣ®дёӯзҡ„jsжЁЎеқ—еҸҜд»Ҙз”Ё unplugin-auto-import жқҘиҮӘеҠЁеј•е…ҘгҖӮ

  1. йЎ№зӣ®жң¬ең°зј–еҶҷзҡ„дёҖдәӣе…·ж–№жі•иҮӘеҠЁеј•е…Ҙ
  2. дёҖдәӣз¬¬дёүж–№еә“зҡ„иҮӘеҠЁеј•е…Ҙ

еңЁ vite.config.js ж–Үд»¶дёӯеј•е…ҘпјҢй…ҚзҪ®жҸ’д»¶гҖӮ

js
import AutoImport from 'unplugin-auto-import/vite'

export default defineConfig({
  plugins: [
    vue(),
    AutoImport({
      imports: [
        // auto-importеҶ…зҪ®vueгҖҒvue-routerгҖҒreactиҝҷдәӣеёёи§Ғеҹәжң¬еә“зҡ„еј•е…Ҙи§„еҲҷ
        "vue"
      ]
    })
  ]
})

жҸ’д»¶еҶ…зҪ®дәҶ vue зҡ„и§„еҲҷпјҢжүҖд»ҘеҸҜд»ҘзӣҙжҺҘеҶҷгҖӮеҗҺз»ӯж–Үд»¶дёӯж— йңҖжүӢеҠЁеј•е…Ҙ ref зӯүжЁЎеқ—пјҢеҸҜзӣҙжҺҘдҪҝз”ЁгҖӮ

еҰӮжһңжҳҜжғіз”ЁиҮӘе·ұе°ҒиЈ…зҡ„еҮҪж•°ж–№жі•пјҢйңҖиҰҒеҒҡеҰӮдёӢй…ҚзҪ®пјҡ

js
import AutoImport from 'unplugin-auto-import/vite'

export default defineConfig({
  plugins: [
    vue(),
    AutoImport({
      imports: [
        // auto-importеҶ…зҪ®vueгҖҒvue-routerгҖҒreactиҝҷдәӣеёёи§Ғеҹәжң¬еә“зҡ„еј•е…Ҙи§„еҲҷ
        'vue',
      ],
      // й…ҚзҪ®йӮЈдәӣжң¬ең°зӣ®еҪ•пјҢж”ҜжҢҒиҮӘеҠЁеј•е…Ҙ
      dirs: [
        	'./src/utils/**'
      ]
    })
  ]
})

еҰӮжһңжғіиҰҒиҮӘеҠЁеј•е…Ҙз¬¬дёүж–№еә“еҰӮ axios гҖҒloadsh зӯүпјҢжІЎжңүеҶ…зҪ®ж”ҜжҢҒзҡ„пјҢеҲҷйңҖиҰҒиҮӘе·ұзј–еҶҷи§„еҲҷгҖӮ

js
import AutoImport from 'unplugin-auto-components/vite'

export default defineConfig({
  plugins: [
    vue(),
    AutoImport({
      imports: [
        'vue',
        {
          // ж”ҫиҮӘе®ҡд№үзҡ„иҮӘеҠЁеј•е…Ҙеә“
          "loadsh": [
            // е“Әдәӣж–№жі•иғҪиҮӘеҠЁеј•е…Ҙ
            "concat"
          ]
        }
      ],
      // й…ҚзҪ®йӮЈдәӣжң¬ең°зӣ®еҪ•пјҢж”ҜжҢҒиҮӘеҠЁеј•е…Ҙ
      dirs: [
        	'./src/utils/**'
      ]
    })
  ]
})

дёҠж–№й…ҚзҪ®жң¬иҙЁдёҠжҳҜйҖҡиҝҮ import {concat} from 'loadsh' еј•е…Ҙзҡ„пјҢеӣ жӯӨеҸҜд»Ҙи®ҫзҪ®еҲ«еҗҚгҖӮ

js
import AutoImport from 'unplugin-auto-components/vite'

export default defineConfig({
  plugins: [
    vue(),
    AutoImport({
      imports: [
        {
          // ж”ҫиҮӘе®ҡд№үзҡ„иҮӘеҠЁеј•е…Ҙеә“
          "loadsh": [
            // е“Әдәӣж–№жі•иғҪиҮӘеҠЁеј•е…Ҙ
            "concat",
            ["compact", "cp"]
          ]
        }
      ]
    })
  ]
})

иҝҷд№Ҳдҝ®ж”№е®ғе°ұдјҡи®ҫзҪ®дёҠеҲ«еҗҚпјҢжң¬иҙЁдёҠжҳҜ import {compact as cp} from 'loadsh' гҖӮ

еғҸ axios иҝҷз§Қе…ЁйғЁеҜје…Ҙй»ҳи®ӨйғЁеҲҶзҡ„пјҢе®һйҷ…дёҠеҜје…Ҙзҡ„жҳҜ default пјҢеӣ жӯӨеҸҜд»ҘйҖҡиҝҮи®ҫзҪ®еҲ«еҗҚзҡ„ж–№ејҸиҮӘеҠЁеј•е…ҘгҖӮ

js
import AutoImport from 'unplugin-auto-components/vite'

export default defineConfig({
  plugins: [
    vue(),
    AutoImport({
      imports: [
        {
          // ж”ҫиҮӘе®ҡд№үзҡ„иҮӘеҠЁеј•е…Ҙеә“
          "loadsh": [
            // е“Әдәӣж–№жі•иғҪиҮӘеҠЁеј•е…Ҙ

жңҖз»Ҳзј–иҜ‘еҮәжқҘзҡ„ж•Ҳжһңзӯүд»·дәҺ import axios from 'axios' гҖӮ

unplugin-vue-components вҖӢ

VueйЎ№зӣ®з»„д»¶ unplugin-auto-import иҷҪеҸҜз”ЁжқҘиҮӘеҠЁеј•е…ҘпјҢдҪҶеҫҲеӨҡ vue йЎ№зӣ®з»„д»¶жҳҜз”Ё .vue ж–Үд»¶зј–еҶҷпјҢunpluin-auto-import еҒҡдёҚеҲ°иҮӘеҠЁеј•е…ҘпјҢжүҖд»Ҙжңүиҝҷд№ҲдёҖдёӘдё“й—Ёдёә vue еҮҶеӨҮзҡ„иҮӘеҠЁеј•е…Ҙеә“гҖӮиҖҢ react зәҜзІ№зҡ„е°ұжҳҜ js пјҢжүҖд»Ҙ react еҸҜд»ҘзӣҙжҺҘдҪҝз”Ё unplugin-auto-import гҖӮ

  1. еҰӮдҪ•еј•е…ҘиҮӘеҠЁжң¬ең°з»„д»¶
  2. иҮӘе·ұејҖеҸ‘зҡ„з»„д»¶еә“еҰӮдҪ•еҒҡеҲ°иҮӘеҠЁж”ҜжҢҒиҮӘеҠЁеј•е…Ҙ

еҗҢж ·зҡ„пјҢйңҖиҰҒе…ҲеҺ» vite.config.js ж–Үд»¶пјҢеҒҡзӣёе…ій…ҚзҪ®пјҢд»Ҙ element-plus дёҫдҫӢгҖӮ

js
import Components from 'unplugin-vue-components/vite'
import { ElementPlusResolver } from 'unplugin-vue-components/resolvers'

export default defineConfig({
  plugins: [
    vue(),
    Components({
      // дҫҝжҗәз»„д»¶еә“зҡ„иҮӘеҠЁеј•е…Ҙи§„еҲҷгҖӮеҶ…зҪ®дәҶдё»жөҒзҡ„з»„д»¶пјҢеҰӮelementгҖҒantdгҖҒvantзӯү
      resolvers: [
        ElementPlusResolver()
      ]
    })
  ]
})

жң¬ең°з»„д»¶д№ҹжғіиҰҒе®һзҺ°иҮӘеҠЁеј•е…ҘпјҢе’Ң unplugin-auto дёҖж ·йҖҡиҝҮ dirs й…ҚзҪ®гҖӮ

js
import Components from 'unplugin-vue-components/vite'
import { ElementPlusResolver } from 'unplugin-vue-components/resolvers'

export default defineConfig({
  plugins: [
    vue(),
    Components({
      // дҫҝжҗәз»„д»¶еә“зҡ„иҮӘеҠЁеј•е…Ҙи§„еҲҷгҖӮеҶ…зҪ®дәҶдё»жөҒзҡ„з»„д»¶пјҢеҰӮelementгҖҒantdгҖҒvantзӯү
      resolvers: [
        ElementPlusResolver()
      ],
      // еӯҳж”ҫз»„д»¶зҡ„ж–Үд»¶еӨ№
      dirs: [
        './src/components',
      ]
    })
  ]
})

еҰӮжһңжғіиҰҒиҮӘе®ҡд№ү resolver жқҘж”ҜжҢҒиҮӘдё»ејҖеҸ‘зҡ„з»„д»¶еә“пјҢеҲҷйңҖиҰҒй…ҚзҪ®дёҖдёӘж–№жі•пјҢиҜҘж–№жі•еҸҜжҺҘж”¶жүҖдҪҝз”Ёзҡ„з»„д»¶еҗҚз§°гҖӮеҲӨж–ӯеҗҚз§°жҳҜеҗҰжҳҜиҮӘдё»ејҖеҸ‘зҡ„з»„д»¶пјҲжңүеӣәе®ҡзҡ„е‘ҪеҗҚпјүпјҢз¬ҰеҗҲжқЎд»¶ return дёҖдёӘеҜ№иұЎгҖӮиҜҘеҜ№иұЎдё»иҰҒеұһжҖ§дёәпјҡ

  • Nameпјҡз»„д»¶еҗҚз§°
  • fromпјҡжқҘжәҗж–Үд»¶еӨ№и·Ҝеҫ„
  • styleпјҡж ·ејҸж–Үд»¶и·Ҝеҫ„
js
import Components from 'unplugin-vue-components/vite'
import { ElementPlusResolver } from 'unplugin-vue-components/resolvers'

export default defineConfig({
  plugins: [
    vue(),
    Components({
      // дҫҝжҗәз»„д»¶еә“зҡ„иҮӘеҠЁеј•е…Ҙи§„еҲҷгҖӮеҶ…зҪ®дәҶдё»жөҒзҡ„з»„д»¶пјҢеҰӮelementгҖҒantdгҖҒvantзӯү
      resolvers: [
        ElementPlusResolver(),
        (componentName) => {
          if(componentName.indexOf('My') != -1) {
            // еј•е…Ҙи§„еҲҷ
            return {
              name: componentName,
              from: 'myui',
              sideEffects: 'myui/' + componentName + '/style/index.css'
            }
          }
        }
      ],
      // еӯҳж”ҫз»„д»¶зҡ„ж–Үд»¶еӨ№
      dirs: [
        './src/components',
      ]
    })
  ]
})

еҗҺз»ӯзј–иҜ‘еҗҺзӯүд»·дәҺеј•е…Ҙз»„д»¶е’Ңж ·ејҸж–Үд»¶гҖӮ

js
import {MyButton} from 'myui'
import 'myui/MyButton/style/index.css'

иӢҘжғіиҰҒдәҶи§ЈеҰӮдҪ•ејҖеҸ‘иҮӘе·ұзҡ„з»„д»¶еә“пјҢжҢҮи·Ҝи·іиҪ¬пјҡз»„д»¶еә“жҗӯе»ә гҖӮ
