ProxyдёҺdefineProperty вҖӢ

ProxyдёҺdefinePropertyзҡ„еҢәеҲ« вҖӢ

defineProperty вҖӢ

defineProperty жҳҜ Object еҶ…йғЁеӯҳеңЁзҡ„дёҖдёӘж–№жі•пјҢеҶ…йғЁзҡ„еҹәжң¬ж“ҚдҪңпјҢиғҪеӨҹзӣҙжҺҘеңЁдёҖдёӘеҜ№иұЎдёҠе®ҡд№үдёҖдёӘж–°еұһжҖ§пјҢйҖҡиҝҮ getter иҜ»еҸ–пјҢsetter дҝ®ж”№дёҖдёӘеҜ№иұЎзҡ„зҺ°жңүеұһжҖ§пјҢе№¶иҝ”еӣһжӯӨеҜ№иұЎгҖӮ

Proxy вҖӢ

Proxy жҳҜеңЁ ES6 дёӯж–°еҮәзҡ„дёҖдёӘеҜ№иұЎпјҢе®ғз”ЁдәҺе®ҡд№үеҹәжң¬ж“ҚдҪңзҡ„иҮӘе®ҡд№үиЎҢдёәпјҲеҰӮеұһжҖ§жҹҘжүҫгҖҒиөӢеҖјгҖҒжһҡдёҫгҖҒеҮҪж•°и°ғз”ЁзӯүпјүпјҢз”ЁдәҺеҲӣе»әдёҖдёӘеҜ№иұЎзҡ„д»ЈзҗҶпјҢд»ҺиҖҢе®һзҺ°еҹәжң¬ж“ҚдҪңзҡ„жӢҰжҲӘе’ҢиҮӘе®ҡд№үпјҲеҰӮеұһжҖ§жҹҘжүҫгҖҒиөӢеҖјгҖҒжһҡдёҫгҖҒж–№жі•и°ғз”ЁзӯүпјүгҖӮ

дҪҶдёҚд»…д»…дәҺжӯӨпјҢProxy еҸҜд»ҘжӢҰжҲӘе’ҢйҮҚж–°е®ҡд№үеҜ№иұЎзҡ„еҹәжң¬ж“ҚдҪңгҖӮдҪ•дёәеҜ№иұЎзҡ„еҹәжң¬ж“ҚдҪңпјҹ

еҜ№иұЎзҡ„еҹәжң¬ж“ҚдҪңеҢ…жӢ¬д»ҘдёӢеҮ зӮ№пјҡ

  • еұһжҖ§жҹҘжүҫ
  • еұһжҖ§иөӢеҖј
  • еұһжҖ§еҲ йҷӨ
  • еұһжҖ§жһҡдёҫ
  • ж–№жі•и°ғз”Ё
js
let obj = {};
console.log(obj.a);
obj.b = 2;
'c' in obj;
delete obj.a;
for(const k in obj) {
    console.log(k);
}
obj.c();

иҝҷдәӣеҶҷжі•жҳҜдёәдәҶж–№дҫҝејҖеҸ‘иҖ…зҡ„з®ҖдҫҝеҶҷжі•пјҢе…¶еҶ…йғЁдјҡи°ғз”ЁеҜ№еә”зҡ„еҮҪж•°гҖӮеҰӮиҜ»еҖјж“ҚдҪң obj.a пјҢе®ғеҶ…йғЁдјҡи°ғз”ЁдёҖдёӘ get еҮҪж•°пјӣиөӢеҖјж“ҚдҪң obj.b = 2 пјҢе®ғеҶ…йғЁдјҡи°ғз”ЁдёҖдёӘ set еҮҪж•°пјӣжҹҘжүҫж“ҚдҪң 'c' in obj пјҢе®ғеҶ…йғЁдјҡи°ғз”ЁдёҖдёӘ has еҮҪж•°пјӣеҲ йҷӨж“ҚдҪң delete obj.a пјҢе®ғеҶ…йғЁдјҡи°ғз”Ё delete еҮҪж•°пјӣжһҡдёҫж“ҚдҪң for(const k in obj) пјҢе®ғеҶ…йғЁдјҡи°ғз”ЁдёҖдёӘ ownKeys еҮҪж•°гҖӮ

еҺҹжң¬иҝҷдәӣеҹәжң¬ж“ҚдҪңжҳҜеҶ…йғЁзҡ„гҖҒдёҚдјҡжҡҙйңІз»ҷејҖеҸ‘иҖ…зҡ„пјҢжңүдәҶ Proxy д№ӢеҗҺпјҢејҖеҸ‘иҖ…еҸҜд»ҘжӢҰжҲӘе№¶иҮӘе®ҡд№үиҝҷдәӣеҹәжң¬ж“ҚдҪңгҖӮеҗҺз»ӯеңЁж“ҚдҪңж—¶дёҚдјҡи°ғз”ЁеҶ…йғЁзҡ„ж–№жі•пјҢиҖҢжҳҜи°ғз”ЁејҖеҸ‘иҖ…еңЁ Proxy иҮӘе®ҡд№үзҡ„ж–№жі•гҖӮ

жҖ»з»“ вҖӢ

defineProperty еҺҹжң¬жҳҜеҜ№иұЎзҡ„еҹәжң¬ж“ҚдҪңд№ӢдёҖпјҢеҸӘиғҪз”ЁжқҘе®ҡд№үеұһжҖ§жҸҸиҝ°з¬ҰпјӣProxy еҸҜд»Ҙй’ҲеҜ№жүҖжңүзҡ„еұһжҖ§иҝӣиЎҢжӢҰжҲӘгҖӮ

ProxyжҜ”definePropertyеҘҪеңЁе“Ә вҖӢ

еңЁеӯҰд№  Vue ж—¶пјҢдёҖзӣҙйғҪеҗ¬иҜҙ Vue3 е“Қеә”ејҸеҺҹзҗҶз”Ёзҡ„ ProxyпјҢVue2 дҪҝз”Ёзҡ„жҳҜ definePropertyгҖӮеӣ жӯӨ Vue3 е“Қеә”ејҸжӣҙдјҳдәҺ Vue2гҖӮдёҚиҝҮиҝҷжҳҜдёәд»Җд№Ҳе‘ўпјҹ

Vue2е“Қеә”ејҸжң¬иҙЁ вҖӢ

йҰ–е…Ҳе…ідәҺ Vue зҡ„е“Қеә”ејҸжң¬иҙЁжҳҜд»Җд№ҲпјҢжҳҜеңЁз»ҷдёҖдёӘеҸҳйҮҸйҮҚж–°иөӢеҖјж—¶иғҪеӨҹеҫ—зҹҘеҲ°иҝҷжӯҘж“ҚдҪңпјҢеҒҡжҹҗдәӣеӨ„зҗҶпјҢе№¶жӣҙж–°йЎөйқўгҖӮ

js
const obj = {
    a: 1,
    b: 2,
    c: {
        tree: 1,
        flower: 2
    }
}

зҺ°еңЁжҲ‘иӢҘзӣҙжҺҘ obj.a = 2 дҝ®ж”№еҜ№иұЎ obj зҡ„ a еұһжҖ§пјҢжҳҫз„¶жҳҜж— жі•еҒҡеӨ„зҗҶзҡ„гҖӮиӢҘдҪҝз”ЁеҮҪж•°зҡ„ж–№ејҸе‘ўпјҹйҖҡиҝҮ get() иҺ·еҸ–еҖјпјҢйҖҡиҝҮ set() и®ҫзҪ®еҖјгҖӮ

дёҚиҝҮжҖҺд№ҲеҸҳжҲҗеҮҪж•°е‘ўпјҹеңЁ ES6 д№ӢеүҚпјҢеҸӘиғҪйҖҡиҝҮ Object.defineProperty е®һзҺ°гҖӮ

js
Object.defineProperty(obj, 'a', {
    get() {
        console.log('иҜ»еҸ–a:', a)
        return a
    },
    set(val) {
        if(val !== obj.a) {
            console.log('жӣҙж”№a:', a)
            obj.a = val
        }
    }
})

иҝҷж ·е°ұиғҪеӨҹеңЁиҺ·еҸ– a ж—¶зӣ‘еҗ¬еҲ°пјҢеңЁжӣҙж”№ a ж—¶д№ҹиғҪзӣ‘еҗ¬еҲ°пјҢиҝҳиғҪеҒҡе…¶д»–зҡ„ж“ҚдҪңгҖӮ

з”ұдәҺ Object.defineProperty жҳҜй’ҲеҜ№еұһжҖ§еҒҡзӣ‘еҗ¬пјҢжүҖд»ҘеҜ№дәҺдёҖдёӘеҜ№иұЎе®ғйңҖиҰҒеҒҡж·ұеәҰйҒҚеҺҶпјҢеңЁ Vue2 жәҗз ҒдёӯпјҢе°ұжңүдёҖдёӘ observe зҡ„еҮҪж•°пјҢйҒҚеҺҶдј иҝҮжқҘзҡ„еҜ№иұЎиҝӣиЎҢзӣ‘еҗ¬гҖӮеҰӮжһңиҜҘеұһжҖ§еҖјжҳҜдёҖдёӘеҜ№иұЎпјҢеҲҷеҶҚдёҖж¬Ўж·ұеәҰйҒҚеҺҶгҖӮ

js
function _isObject(v) {
    return typeof v === 'object' && v !== null
}

function observe(obj) {
    for(const k in obj) {
        let v = obj[k]
        if(_isObject(v)) observe(v)
    }
    Object.defineProperty(obj, k, {
        get() {
            console.log('иҜ»еҸ–:', k)
            return v
        },
        set(val) {
            if(val !== v) {
                console.log('жӣҙж”№:', k)
                v = val
            }
        }
    })
}

иҝҷдёӘиҝҮзЁӢе°ұеҸ«еҒҡи§ӮеҜҹпјҢи§ӮеҜҹе®һиҙЁе°ұжҳҜйҒҚеҺҶжҜҸдёӘеұһжҖ§пјҢжҠҠжҜҸдёҖдёӘеұһжҖ§зҡ„иҜ»еҸ–е’ҢиөӢеҖјеҸҳжҲҗеҮҪж•°пјҢиҝҷж ·е°ұиғҪж”¶еҲ°йҖҡзҹҘгҖӮ

иҝҷдёӘеҒҡжі•жңүдёҖдёӘеӨ©з”ҹзҡ„зјәйҷ·пјҡз”ұдәҺе®ғеҸӘиғҪеҜ№еұһжҖ§зӣ‘еҗ¬пјҢеӣ жӯӨйңҖиҰҒж·ұеәҰйҒҚеҺҶгҖӮдё”зӣ‘еҗ¬зҡ„ж—¶еҖҷеңЁйҒҚеҺҶжҳҜиғҪзӣ‘еҗ¬еҜ№иұЎзҡ„еҺҹеұһжҖ§пјҢеҰӮжһңж–°еўһдёҖдёӘеұһжҖ§пјҢеҲҷж— жі•иў«зӣ‘еҗ¬гҖӮиҝҷе°ұжҳҜдёәд»Җд№Ҳ Vue2 ж— жі•зӣ‘еҗ¬еҜ№иұЎеұһжҖ§зҡ„ж–°еўһе’ҢеҲ йҷӨпјҲдёҚдјҡи§ҰеҸ‘ get е’Ң set пјүгҖӮ

Vue3е“Қеә”ејҸжң¬иҙЁ вҖӢ

еҗҢж ·йғҪжҳҜеҸҳеҮҪж•°пјҢVue3 жҳҜзӣҙжҺҘй’ҲеҜ№ж•ҙдёӘеҜ№иұЎиҝӣиЎҢзӣ‘еҗ¬гҖӮж— и®әжҳҜж–°еўһгҖҒеҲ йҷӨиҝҳжҳҜиҜ»еҸ–пјҢд»–йғҪиғҪзӣ‘еҗ¬еҲ°гҖӮ

еңЁзӣ‘еҗ¬ж—¶д»–иҝҳдјҡдә§з”ҹдёҖдёӘд»ЈзҗҶеҜ№иұЎпјҢеңЁдҪҝз”ЁеұһжҖ§жҲ–иөӢеҖјеұһжҖ§ж—¶пјҢйғҪжҳҜйҖҡиҝҮиҝҷдёӘд»ЈзҗҶеҜ№иұЎжқҘе®һзҺ°гҖӮ

js
const newObj = new Proxy(obj, {
    get(target, k) {
        let v = target[k]
        console.log(k, 'read')
        return v
    },
    set(target, k, val) {
        let v = target[k]
        if(target[k] !== val) {
            target[k] = val
        }
    }
})

е®Ңе–„дёҖдёӢгҖӮ

js
function _isObject(v) {
    return typeof v === 'object' && v !== null
}

function observe(obj) {
    const newObj = new Proxy(obj, {
        get(target, k) {
            let v = target[k]
            if(_isObject(v)) {
                v = observe(v)
            }
            console.log(k, 'read')
            return v
        },
        set(target, k, val) {
            let v = target[k]
            if(target[k] !== val) {
                target[k] = val
            }
        }
    })
    return newObj
}

жҖ»з»“ вҖӢ

з”ұдәҺ proxy жҳҜзӣҙжҺҘзӣ‘еҗ¬ж•ҙдёӘеҜ№иұЎпјҢеңЁйңҖиҰҒзҡ„ж—¶еҖҷжүҚдјҡйҖ’еҪ’зӣ‘еҗ¬еҜ№иұЎеҶ…зҡ„еҜ№иұЎпјҢеӣ жӯӨжҖ§иғҪдёҠдјҡжҜ” defineProporty жӣҙиғңдёҖзӯ№гҖӮ
